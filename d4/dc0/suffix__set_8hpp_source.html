99 if (this->endpoint()) // проверка на краевые точки

101 if (this->empty()) // если дерево пусто (нет корня)

111#ifdef PATL_DEBUG

113#else

116#endif

120 word_t count_reindex(const key_type from) const
121 {
122 key_type last = this->keys_ + this->size() * this->delta;
123 word_t i = this->size() - 1;
124 for (
125 ; last + this->trie_[i]->get_skip() / this->bit_size >= from
126 ; --i, last -= this->delta) ;
127 return this->size() - 1 - i;
128 }
