16// L - lenght of sorting buffer
17// dbuff - buffer of lenght 2*L+1, source data in first half
18// in result we have sorted data in first half of dbuff
19// return value - position of zero symbol
20static unsigned bwtSort(unsigned L, unsigned char *dbuff, SuffixSet &sufSet, unsigned &mult)
21{
22 // copy 1st half into 2nd
23 memcpy(dbuff + L, dbuff, L);
24 // stopbyte
25 dbuff[2 * L] = ~dbuff[0];
26 //
27 for (;;)
28 {
29 // add next node and check for loop
30 const SuffixSet::vertex v = sufSet.push_back();
31 const unsigned skip = v.skip();
32 if (sufSet.size() - 1 + skip / 8 == 2 * L)
33 {
34 sufSet.pop_back();
35 mult = L / sufSet.size();
36 break;
37 }
38 }
39 //
40 SuffixSet::const_iterator cit = sufSet.begin();
41 unsigned id = 0, offs;
42 for (; (offs = *cit - dbuff) != 0; ++cit)
43 dbuff[id++] = dbuff[offs + L - 1];
44 const unsigned zeroId = id;
45 for (; cit != sufSet.end(); ++cit)
46 dbuff[id++] = dbuff[(*cit - dbuff) + L - 1];
47 return zeroId;
48}

50// L - lenght of unsorting buffer
51// dbuff - buffer of lenght 2*L, source data in first half
52// pos - position of zero symbol (propagate zeroId)
53// vect - allocated memory for storing backward transform vector, size (0x100+L)
54// we have resulting data in second half of dbuff
55static void bwtUnsort(unsigned L, unsigned char *dbuff, unsigned pos, unsigned *vect)
56{
57 unsigned *freq = vect;
58 vect += 0x100;
59 std::fill(freq, freq + 0x100, 0);
60 unsigned id;
61 for (id = 0; id != L; ++id)
62 ++freq[dbuff[id]];
63 unsigned sum = 0;
64 for (id = 0; id != 0x100; ++id)
65 freq[id] = (sum += freq[id]) - freq[id];
66 for (id = 0; id != L; ++id)
67 vect[freq[dbuff[id]]++] = id;
68 for (id = 0; id != L; ++id)
69 dbuff[L + id] = dbuff[pos = vect[pos]];
70}

72static inline unsigned readSize(const std::string &str)
73{
74 unsigned num = 0;
75 std::string::const_iterator it = str.begin();
76 for (; it != str.end() && '0' <= *it && *it <= '9'; ++it)
77 {
78 num *= 10;
79 num += *it - '0';
80 }
81 if (it != str.end())
82 {
83 switch (*it)
84 {
85 case 'm':
86 num <<= 10;
87 case 'k':
88 num <<= 10;
89 break;
90 }
91 }
92 return num;
93}

107 argv[1][0] == 't' || argv[1][0] == 'T' ? 2 : -1;

111 if (opId != -1)
112 {

123 blockSize = readSize(argv[4]);

125 //
