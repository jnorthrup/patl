21class win32_exception
22 : public std::exception
23{
24public:
25 win32_exception(const char *fun_name, const char *fname, int loc)
26 : mess_(64 << 10)
27 {
28 const DWORD mess_id(::GetLastError());
29 char msg_buf[1024];
30 ::FormatMessage(
31 FORMAT_MESSAGE_FROM_SYSTEM,
32 0,
33 mess_id,
34 MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
35 msg_buf,
36 1024,
37 0);
38 sprintf(&mess_[0], "%s (%d) : code %d in %s: %s",
39 fname,
40 loc,
41 mess_id,
42 fun_name,
43 msg_buf);
44 }
45
46 ~win32_exception() throw()
47 {
48 }
49
50 const char *what() const throw()
51 {
52 return &mess_[0];
53 }
54
55private:
56 std::vector<char> mess_;
57};

85inline word_t get_file_length(HANDLE fh)
86{
87 const word_t cur = ::SetFilePointer(fh, 0, 0, FILE_CURRENT);
88 const word_t len = ::SetFilePointer(fh, 0, 0, FILE_END);
89 if (cur == INVALID_SET_FILE_POINTER || len == INVALID_SET_FILE_POINTER ||
90 ::SetFilePointer(fh, cur, 0, FILE_BEGIN) == INVALID_SET_FILE_POINTER)
91 PATL_WIN32_EXCEPTION(SetFilePointer);
92 return len;
93}

95inline long long get_long_file_length(HANDLE fh)
96{
97 LONG cur_h = 0, len_h = 0;
98 const DWORD cur = ::SetFilePointer(fh, 0, &cur_h, FILE_CURRENT);
99 const DWORD len = ::SetFilePointer(fh, 0, &len_h, FILE_END);
100 if (cur == INVALID_SET_FILE_POINTER || len == INVALID_SET_FILE_POINTER ||
101 ::SetFilePointer(fh, cur, &cur_h, FILE_BEGIN) == INVALID_SET_FILE_POINTER)
102 PATL_WIN32_EXCEPTION(SetFilePointer);
103 return
104 static_cast<long long>(len) |
105 (static_cast<long long>(len_h) << (8 * sizeof(long)));
106}

108inline long long get_long_file_length(const char *fname)
109{
110 WIN32_FIND_DATA data;
111 const HANDLE ffh = ::FindFirstFile(fname, &data);
112 if (ffh == INVALID_HANDLE_VALUE)
113 PATL_WIN32_EXCEPTION(FindFirstFile);
114 ::FindClose(ffh);
115 return
116 static_cast<long long>(data.nFileSizeLow) |
117 (static_cast<long long>(data.nFileSizeHigh) << (8 * sizeof(long)));
118}

120inline std::string take_file_folder(const std::string &str)
121{
122 const std::string::size_type n = str.find_last_of("\\/:");
123 if (n == std::string::npos)
124 return ".";
125 switch (str[n])
126 {
127 case ':':
128 return str.substr(0, n + 1);
129 case '\\':
130 case '/':
131 if (str[n - 1] == ':')
132 return str.substr(0, n + 1);
133 else
134 return str.substr(0, n);
135 default:
136 return "error";
137 }
138}

140inline std::string take_file_name_ext(const std::string &str)
141{
142 const std::string::size_type n = str.find_last_of("\\/:");
143 return str.substr(0, n);
144}
