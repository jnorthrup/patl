383 iterator insert(iterator hint, const value_type &val)
384 {
385 if (this->root_)
386 {
387 algorithm &pal = static_cast<algorithm&>(static_cast<vertex&>(hint));
388 const key_type &key = T::ref_key(val);
389 const word_t skip = this->bit_comp_.bit_mismatch(key, pal.get_key());
390 if (~word_t(0) == skip)
391 return iterator(vertex(pal));
392 pal.ascend_less(skip);
393 // find a number of first mismatching bit
394 {
395 const word_t len = this->bit_comp_.bit_length(key);
396 if (len == ~word_t(0))
397 pal.run(key);
398 else
399 pal.run(key, len);
400 }
401 const word_t l =
402 this->bit_comp_.bit_mismatch(key, pal.get_key(), skip);
403 // if this number end at infinity
404 if (~word_t(0) == l)
405 // then this key already in trie
406 return iterator(vertex(pal));
407 else
408 {
409 pal.ascend(l);
410 // add new node for value with unique key
411 return iterator(vertex(add(val, pal, l)));
412 }
413 }
414 // if trie is empty then add root

435 template <typename Handler>
436 void merge(const_iterator first, const_iterator last, Handler handler)
437 {
438 // if trie is empty it is necessary to insert first element
439 if (!this->root_ && first != last)
440 {
441 insert(*first);
442 ++first;
443 }
444 algorithm pal_cur(this, this->root_, 0);
445 const_vertex vtx(first);
446 const algorithm &pal_end(static_cast<const const_vertex&>(last));
447 word_t skip = 0;
448 while (static_cast<const algorithm&>(vtx) != pal_end)
449 {
450 pal_cur.ascend_less(skip);
451 const word_t l = pal_cur.mismatch(vtx.key());
452 if (~word_t(0) == l)
453 // identical keys found - handler must be applied
454 handler(iterator(vertex(pal_cur)), const_iterator(vtx));
455 else
456 add(vtx.value(), pal_cur, l);
457 // move to the next
458 if (vtx.get_qid())
459 {
460 const node_type *cur = static_cast<const algorithm&>(vtx).get_q();
461 for (; cur->get_parent_id(); cur = cur->get_parent()) ;
462 vtx = vertex(this, cur->get_parent(), 1);
463 }
464 else
465 vtx.toggle();
466 skip = static_cast<const algorithm&>(vtx).get_q()->get_skip();
467 vtx.template descend<0>();
468 }
469 }
470
471 void merge(const_iterator first, const_iterator last)
472 {
473 merge(first, last, void_merge_handler());
474 }

487 template <typename OutIter, typename ShowValue>
488 void show(OutIter oit, ShowValue show_val) const
489 {
490 if (!this->root_)
491 return;
492 const_vertex vtx(this->root());
493 show_val(oit, vtx.parent_key());
494 word_t skip_ins = 0;
495 for (const_preorder_iterator pit = vtx.preorder_begin()
496 ; pit != vtx.preorder_end()
497 ; ++pit)
498 {
499 if (pit->get_qtag())
500 ++skip_ins;
501 else
502 {
503 if (skip_ins)
504 {
505 *oit++ = highest_bit | skip_ins;
506 skip_ins = 0;
507 }
508 *oit++ = pit->next_skip();
509 show_val(oit, pit->key());
510 }
511 }
512 }

521 struct default_read_value
522 : public std::unary_function<InIter, value_type>

531 template <typename InIter, typename ReadValue>
532 void read(InIter iit, InIter iit_end, ReadValue read_val)
533 {
534 clear();
535 if (iit == iit_end)
536 return;
537 preorder_iterator pit(vertex(add_root(read_val(iit))));
538 while (iit != iit_end)
539 {
540 const word_t skip = *iit++;
541 if (skip & highest_bit)
542 std::advance(pit, bits_but_highest(skip));
543 else
544 {
545 static_cast<algorithm&>(*pit) =
546 add(read_val(iit), *pit, skip);
547 if (pit->get_qid())
548 pit->toggle();
549 }
550 }
551 }

566 size_type erase(const key_type &key, word_t prefixLen = ~word_t(0))
567 {
568 if (this->root_)
569 {
570 algorithm pal(this, this->root_, 0);
571 if (pal.mismatch(key, prefixLen) >= prefixLen)
572 {
573 const size_type pastSize = size();
574 // if erase entire tree
575 if (pal.get_q() == this->root_)
576 {
577 clear();
578 return pastSize;
579 }
580 erase_subtree(pal);
581 return pastSize - size();
582 }
583 }
584 return 0;
585 }

613 // if erase subtree

616 // deallocate subtree
617 del_tree(pal.get_p());
618 // need to find the node with key whose prefix
619 // we erase among nodes from pal.get_q() up to root_

629 // special case: del root
