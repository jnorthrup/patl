34inline word_t get_lowest_bit_id(word_t x)
35{
36 PATL_ASSERT(x);
37#ifdef PATL_INTRINSIC
38
39 unsigned long r;
40#ifdef PATL_64
41 _BitScanForward64(&r, x);
42#endif // PATL_64
43#ifdef PATL_32
44 _BitScanForward(&r, x);
45#endif // PATL_32
46
47#else // PATL_INTRINSIC
48
49 word_t r = 0;
50 x &= word_t(0) - x; // isolate lowest bit
51#ifdef PATL_64
52 if (x & 0xFFFFFFFF00000000) r += 32;
53 if (x & 0xFFFF0000FFFF0000) r += 16;
54 if (x & 0xFF00FF00FF00FF00) r += 8;
55 if (x & 0xF0F0F0F0F0F0F0F0) r += 4;
56 if (x & 0xCCCCCCCCCCCCCCCC) r += 2;
57 if (x & 0xAAAAAAAAAAAAAAAA) ++r;
58#endif // PATL_64
59#ifdef PATL_32
60 if (x & 0xFFFF0000) r += 16;
61 if (x & 0xFF00FF00) r += 8;
62 if (x & 0xF0F0F0F0) r += 4;
63 if (x & 0xCCCCCCCC) r += 2;
64 if (x & 0xAAAAAAAA) ++r;
65#endif // PATL_32
66
67#endif // PATL_INTRINSIC
68 return r;
69}

71inline word_t get_highest_bit_id(word_t x)
72{
73 PATL_ASSERT(x);
74#ifdef PATL_INTRINSIC
75
76 unsigned long r;
77#ifdef PATL_64
78 _BitScanReverse64(&r, x);
79#endif // PATL_64
80#ifdef PATL_32
81 _BitScanReverse(&r, x);
82#endif // PATL_32
83
84#else // PATL_INTRINSIC
85
86 word_t r = 0;
87#ifdef PATL_64
88 if (x & 0xFFFFFFFF00000000) { x >>= 32; r += 32; }
89#endif // PATL_64
90 if (x & 0xFFFF0000) { x >>= 16; r += 16; }
91 if (x & 0x0000FF00) { x >>= 8; r += 8; }
92 if (x & 0x000000F0) { x >>= 4; r += 4; }
93 if (x & 0x0000000C) { x >>= 2; r += 2; }
94 if (x & 0x00000002) ++r;
95
96#endif // PATL_INTRINSIC
97 return r;
98}

112inline word_t get_binary_lca(word_t rootH, word_t i, word_t j)
113{
114 if (i == j)
115 return i;
116 //sort_2(i, j); // à íóæíî ëè?
117 const word_t
118 iDeep = rootH - get_lowest_bit_id(i),
119 jDeep = rootH - get_lowest_bit_id(j),
120 xDeep = rootH - get_highest_bit_id(i ^ j);
121 if (xDeep > get_min(iDeep, jDeep))
122 return iDeep < jDeep ? i : j;
123 const word_t sh = rootH - xDeep;
124 return (i >> sh | word_t(1)) << sh;
125}
