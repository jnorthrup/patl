199 word_t mismatch_suffix(
200 const key_type key,
202 word_t skip)
203 {
204 const bit_compare bcmp = bit_comp();
205 // until we achieved the leaf
206 while (!get_qtag())
207 {
208 const word_t skip_cur = get_p()->get_skip();
209 if (skip <= skip_cur)
210 {
211 const key_type exist_key = CSELF->get_key();
212 for (; skip != skip_cur; ++skip)
213 {
214 // if we discover mismatch in bits - immediately return its number
215 if (bcmp.get_bit(key, skip) != bcmp.get_bit(exist_key, skip))
216 return skip;
217 }
218 ++skip;
219 }
220 iterate_by_key(key);
221 }
222 return align_down<bit_compare::bit_size>(skip) +
223 bcmp.bit_mismatch(
224 key + skip / bit_compare::bit_size,
225 CSELF->get_key() + skip / bit_compare::bit_size);
226 }

307 node_type *erase()
308 {
309 // p-node back-referenced from q-node (q-tag == 1)
310 // q- & p-node may be the same
311 node_type
312 *q = get_q(),
313 *p = get_p();
314 // p-brother is the sibling of p-node
315 const word_t
316 p_brother_id = word_t(1) ^ get_qid(),
317 p_brother_tag = q->get_xtag(p_brother_id);
318 node_type *p_brother = CSELF->get_xlink(q, p_brother_id);
319 // take the parent of q-node
320 const word_t q_parent_id = q->get_parent_id();
321 node_type *q_parent = CSELF->get_parent(q);
322 // if q-parent exist i.e. q-node isn't the root
323 if (q_parent)
324 {
325 // replace q-node with p-brother
326 CSELF->set_xlinktag(q_parent, q_parent_id, p_brother, p_brother_tag);
327 if (!p_brother_tag)
328 CSELF->set_parentid(p_brother, q_parent, q_parent_id);
329 }
330 // if p- & q-nodes is not the same
331 if (q != p)
332 {
333 // replace p-node with q-node
334 q->set_all_but_value(p);
335 // correct p-parent, if one exist
336 node_type *p_parent = CSELF->get_parent(p);
337 if (p_parent)
338 CSELF->set_xlinktag(p_parent, p->get_parent_id(), q, 0);
339 // correct p-node children
340 if (!p->get_xtag(0))
341 CSELF->set_parentid(CSELF->get_xlink(p, 0), q, 0);
342 if (!p->get_xtag(1))
343 CSELF->set_parentid(CSELF->get_xlink(p, 1), q, 1);
344 }
345 //
346 return p;
347 }

352 word_t prefixLen = ~word_t(0))
353 {

355 const word_t len = get_min(prefixLen, bcmp.bit_length(key));
