46template <typename T>
47void parse_text(patl::aux::buffered_input &inp, T insertIt)
48{
49 bool intoken = false;
50 WordType token;
51 while (inp.is_data())
52 {
53 const unsigned char c = inp.read_byte();
54 if (patl::aux::is_proper_char(c))
55 {
56 if (!intoken)
57 {
58 if (text.size() >= (16 << 20))
59 return;
60 token.clear();
61 intoken = true;
62 }
63 token.push_back(c);
64 }
65 // space between tokens
66 else if (intoken)
67 {
68 insertIt(token);
69 intoken = false;
70 }
71 }
72 if (intoken)
73 insertIt(token);
74}

76template <typename T>
77void parse_fiction(patl::aux::buffered_input &inp, T insertIt)
78{
79 bool
80 inword = false,
81 inpunct = false,
82 isdots = false;
83 WordType curword;
84 while (inp.is_data())
85 {
86 const unsigned char c = patl::aux::to_lower(inp.read_byte());
87 if (patl::aux::is_hyphen(c))
88 {
89 if (inword || inpunct)
90 curword.push_back(c);
91 else
92 {
93 curword.clear();
94 curword.push_back(c);
95 inpunct = true;
96 }
97 }
98 else if (patl::aux::is_letter(c))
99 {
100 if (inword)
101 curword.push_back(c);
102 else
103 {
104 if (inpunct)
105 {
106 if (isdots)
107 insertIt(".");
108 isdots = inpunct = false;
109 }
110 curword.clear();
111 curword.push_back(c);
112 inword = true;
113 }
114 }
115 else if (patl::aux::is_punctuation(c))
116 {
117 if (inpunct)
118 curword.push_back(c);
119 else
120 {
121 if (inword)
122 {
123 insertIt(curword);
124 inword = false;
125 }
126 curword.clear();
127 curword.push_back(c);
128 inpunct = true;
129 }
130 isdots = isdots || patl::aux::is_dot_symbol(c);
131 }
132 else // is_blank_char or else
133 {
134 if (inword)
135 {
136 insertIt(curword);
137 inword = false;
138 }
139 else if (inpunct)
140 {
141 if (isdots)
142 insertIt(".");
143 isdots = inpunct = false;
144 }
145 }
146 }
147 if (inword || inpunct)
148 insertIt(curword);
149}

151inline void insertIt(const WordType &word)
152{
153 unsigned i = words.size();
154 words.push_back(word);
155 WordSet::const_iterator cit = noiseWordset.find(&words[i]);
156 if (cit != noiseWordset.end())

158 words.pop_back();

160 }
161 //wordStdSet.insert(&words[i]); // для проверки правильности
162 std::pair<WordSet::iterator, bool> itIns = wordset.insert(&words[i]);
163 if (!itIns.second)
164 words.pop_back();
165 /*else

168 log2.write_byte('\n');
169 }*/
170 text.push_back(*itIns.first);

173}
