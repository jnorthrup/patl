24template <typename Iter>
25void print_regexp(word_t length, const Iter &beg, const Iter &end)
26{
27 const word_t limit = (length + 1) * 8;
28 for (Iter cur = beg; cur != end; cur.increment(limit))
29 {
30 if (!cur->limited(limit))
31 continue;
32 //
33 if (beg != cur)
34 printf("|");
35 if (cur->get_qtag())
36 printf("%s", cur->key().substr(length).c_str());
37 else
38 {
39 const word_t next_len = cur->next_skip() / 8;
40 printf("%s", cur->key().substr(length, next_len - length).c_str());
41 Iter
42 it1 = cur->preorder_begin((next_len + 1) * 8),
43 it2 = cur->preorder_end();
44 printf("(");
45 const bool question = it1->key().size() == next_len;
46 if (question)
47 ++it1;
48 print_regexp(next_len, it1, it2);
49 printf(")");
50 if (question)
51 printf("?");
52 }
53 }
54}
