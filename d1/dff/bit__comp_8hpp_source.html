24template <typename T>
25class bit_comparator<T, 0>
26{
27public:
28 // optional; used in non-scalar bit_comparator's
29 //typedef undefined char_type;

33
35 word_t bit_length(T) const
36 {
37 return 8 * sizeof(T);
38 }
39
41 word_t get_bit(T val, word_t id) const
42 {
43 return word_t(1) & (impl::unsigned_cast(val) >>

49 {
