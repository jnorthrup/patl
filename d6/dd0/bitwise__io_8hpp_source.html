47 void put_bit(word_t bit)
48 {
49 *cur_ |= (bit & 1) << num_;
50 if (++num_ & bits_in_word)
51 {
52 num_ = 0;
53 flush_full();
54 }
55 }
56
57 void put_bits(word_t bits, word_t n)
58 {
59 const word_t bits0 = bits & (1 << n) - 1;
60 *cur_ |= bits0 << num_;
61 if ((num_ += n) & bits_in_word)
62 {
63 num_ &= bits_in_word - 1;
64 flush_full();
65 *cur_ = bits0 >> (n - num_);
66 }
67 }
68
69 void put_fib_code(word_t n)
70 {
71 word_t i = n < 1597 ? 14 : sizeof(fibs) / sizeof(*fibs) - 1;
72 for (; fibs[i] > n; --i) ;
73 const word_t highest = i + 1;
74 word_t code = 1 << highest;
75 for (; n; --i)
76 {
77 if (fibs[i] <= n)
78 {
79 code |= 1 << i;
80 n -= fibs[i--];
81 }
82 }
83 put_bits(code, highest + 1);
84 }
85
86 void put_word(word_t w)
87 {
88 if (num_)
89 {
90 *cur_ |= w << num_;
91 flush_full();
92 *cur_ = w >> (bits_in_word - num_);
93 }
94 else
95 {
96 *cur_ = w;
97 flush_full();
98 }
99 }
100
101 void put_bytes(const char *bytes, word_t n)
102 {
103 const word_t
104 *cur = reinterpret_cast<const word_t*>(bytes),
105 *end = cur + n / sizeof(word_t);
106 for (; cur != end; ++cur)
107 put_word(*cur);
108 if (n % sizeof(word_t))
109 put_bits(*cur, (n % sizeof(word_t)) * 8);
110 }
111
112private:
113 void flush_full()
114 {
115 if (++cur_ == end_)
116 {
117 const word_t size = (end_ - beg_) * sizeof(word_t);
118 dst_(beg_, size);
119 memset(beg_, 0, size);
120 cur_ = beg_;
121 }
122 }
123
124 void flush()
125 {
126 const word_t size = (cur_ - beg_) * sizeof(word_t) +
127 impl::align_up<8>(num_) / 8;
128 if (size)
129 {
130 dst_(beg_, size);
131 memset(beg_, 0, size);
132 cur_ = beg_;
133 num_ = 0;
134 }
135 }
136
137 Output &dst_;
138 word_t
139 * const beg_,
140 * const end_,
141 *cur_;
142 word_t num_;
143};

145template <typename Input>
146class bit_input

150public:
151 bit_input(Input &src, word_t size = 1 << 16)
152 : src_(src)
153 , beg_(new word_t [size])
154 , end_(beg_ + size)
155 , cur_(beg_)
156 , num_(0)
157 {
158 src_(beg_, size * sizeof(word_t));
159 }
160
161 ~bit_input()
162 {
163 delete[] beg_;
164 }
165
166 word_t get_bit()
167 {
168 const word_t bit = *cur_ >> num_ & 1;
169 if (++num_ & bits_in_word)
170 {
171 num_ = 0;
172 read_full();
173 }
174 return bit;
175 }
176
177 word_t get_bits(word_t n)
178 {
179 word_t bits = *cur_ >> num_;
180 if ((num_ += n) & bits_in_word)
181 {
182 num_ &= bits_in_word - 1;
183 read_full();
184 bits |= *cur_ << (n - num_);
185 }
186 return bits & (1 << n) - 1;
187 }
188
189 word_t get_fib_code()
190 {
191 for (word_t n = 0, i = 0; ; ++i)
192 {
193 if (get_bit())
194 {
195 n += fibs[i++];
196 if (get_bit())
197 return n;
198 }
199 }
200 }
201
202 word_t get_word()
203 {
204 if (num_)
205 {
206 const word_t dword = *cur_ >> num_;
207 read_full();
208 return dword | *cur_ << (bits_in_word - num_);
209 }
210 else
211 {
212 const word_t dword = *cur_;
213 read_full();
214 return dword;
215 }
216 }
217
218 void get_bytes(char *bytes, word_t n)
219 {
220 word_t
221 *cur = reinterpret_cast<word_t*>(bytes),
222 *end = cur + n / sizeof(word_t);
223 for (; cur != end; ++cur)
224 *cur = get_word();
225 if (n % sizeof(word_t))
226 *cur = get_bits((n % sizeof(word_t)) * 8);
227 }
228
229private:
230 void read_full()
231 {
232 if (++cur_ == end_)
233 {
234 src_(beg_, (end_ - beg_) * sizeof(word_t));
235 cur_ = beg_;
236 }
237 }

239 Input &src_;

243 *cur_;
