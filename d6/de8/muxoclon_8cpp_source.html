45void directed_search(
46 const trie_string &dict,
47 vector_wave &wave,
48 vector_wave &antiwave,
49 std::set<const_vertex> &used,
50 bool reversed,
51 vector_vertex &chain)
52{
53 wave.push_back(map_vertex());
54 map_vertex &next = wave.back();
55 const map_vertex
56 &current = wave[wave.size() - 2],
57 &antipode = antiwave.back();
58 for (map_vertex::const_iterator it = current.begin()
59 ; it != current.end()
60 ; ++it)
61 {
62 const const_vertex &seed = it->first;
63 typedef patl::hamming_distance<trie_string, true> hamm_dist;
64 hamm_dist hd(dict, 1, seed.key());
65 trie_string::const_partimator<hamm_dist>
66 pmi = dict.begin(hd),
67 pmi_end = dict.end(hd);
68 for (; pmi != pmi_end; ++pmi)
69 {
70 const const_vertex &vtx = pmi;
71 if (used.find(vtx) == used.end())
72 {
73 const map_vertex::const_iterator match = antipode.find(vtx);
74 if (match != antipode.end())
75 {
76 wave.pop_back();
77 const_vertex cur = reversed ? vtx : seed;
78 vector_wave
79 &front_wave = reversed ? antiwave : wave,
80 &back_wave = reversed ? wave : antiwave;
81 // G++ not allowed to use const_reverse_iterator
82 // error: no match for 'operator!='
83 for (vector_wave::reverse_iterator rit = front_wave.rbegin()
84 ; rit != front_wave.rend()
85 ; cur = rit++->find(cur)->second)
86 chain.push_back(cur);
87 std::reverse(chain.begin(), chain.end());
88 cur = reversed ? seed : vtx;
89 for (vector_wave::reverse_iterator rit = back_wave.rbegin()
90 ; rit != back_wave.rend()
91 ; cur = rit++->find(cur)->second)
92 chain.push_back(cur);
93 return;
94 }
95 used.insert(vtx);
96 next.insert(std::make_pair(vtx, seed));
97 }
98 }
99 }

102bool search_work(
103 const trie_string &dict,
104 const const_vertex &src_vtx,
105 const const_vertex &dst_vtx,
106 vector_vertex &chain)
107{
108 if (src_vtx == dst_vtx)
109 {
110 chain.push_back(src_vtx);
111 return true;
112 }
113 const const_vertex dict_end = dict.end();
114 vector_wave front_wave, back_wave;
115 front_wave.push_back(map_vertex());
116 front_wave.back().insert(std::make_pair(src_vtx, dict_end));
117 back_wave.push_back(map_vertex());
118 back_wave.back().insert(std::make_pair(dst_vtx, dict_end));
119 std::set<const_vertex> front_used, back_used;
120 front_used.insert(src_vtx);
121 back_used.insert(dst_vtx);
122 for (int level = 0; ; ++level)
123 {
124 // front
125 directed_search(dict, front_wave, back_wave, front_used, false, chain);
126 if (!chain.empty())
127 return true;
128 if (front_wave.back().empty())
129 return false;
130 // back
131 directed_search(dict, back_wave, front_wave, back_used, true, chain);
132 if (!chain.empty())
133 return true;
134 if (back_wave.back().empty())
135 return false;
136 }
137}
