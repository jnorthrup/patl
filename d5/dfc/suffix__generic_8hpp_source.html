259 match_iterator(this_t *cont, const key_type &begin)
260 : key_(begin)

290 {

294 word_t skip() const
295 {
296 return skip_;
297 }
298
299 word_t length() const
300 {
301 return skip_ / bit_size;
302 }
303
304 const key_type key() const
305 {
306 return key_;
307 }
308
309 match_iterator &operator++()
310 {
311 skip_ = max0(static_cast<sword_t>(skip_) - delta * bit_size);
312 key_ += delta;
313 const this_t *cont = vtx_.cont();
314 algorithm &pal = static_cast<algorithm&>(vtx_);
315 if (skip_)
316 {
317 const node_type
318 *palP = pal.get_p(),
319 *nextQ = palP == cont->trie_.back()
320 ? cont->root_
321 : cont->trie_.following(palP);
322 pal.init(nextQ, 0);
323 pal.ascend(skip_);
324 const node_type *pretender = pal.get_q();
325 if (pretender != cont->root_)
326 pal.init(
327 pretender,
328 cont->bit_comp_.get_bit(key_, pretender->get_skip()));
329 }
330 else
331 pal.init(cont->root_, 0);
332 //
333 skip_ = pal.mismatch_suffix(key_, skip_);
334 return *this;
335 }
336
337 match_iterator operator++(int)
338 {
339 match_iterator it(*this);
340 ++*this;
341 return it;
342 }
343
344 private:
345 key_type key_;
346 word_t skip_;
347 vertex vtx_;
348 };
349
350protected:
351 vertex push_back_generic(const node_type &init_node)
352 {
353 const key_type key = keys_ + trie_.size() * delta;
354 node_type *back = trie_.back();
355 word_t skip = max0(static_cast<sword_t>(back->get_skip()) - delta * bit_size);
356 algorithm pal(this, this->root_, 0);
357 if (skip)
358 {
359 node_type *back_p = back->get_xlink(back->get_xlink(1) == back ? 0 : 1);
360 pal.init(trie_.following(back_p), 0);
361 pal.ascend(skip);
362 node_type *pretender = pal.get_q();
363 if (pretender != this->root_)
364 pal.init(pretender, this->bit_comp_.get_bit(key, pretender->get_skip()));
365 }
366 //
367 skip = pal.mismatch_suffix(key, skip);
368 // brave new node
369 trie_.push_back(init_node);
370 node_type *r = trie_.back();
371 // add new node into trie
372 const word_t bit = this->bit_comp_.get_bit(key, skip);
373 pal.add(r, bit, skip);
374 return vertex(this, r, bit);
375 }
376
377 vertex push_back_root(const node_type &init_node)
378 {
379 trie_.push_back(init_node);
380 this->root_ = trie_.back();
381 this->root_->init_root();
382 return vertex(this, this->root_, 0);
383 }
384
385 void erase_node(algorithm &pal)
386 {
387 node_type *p = pal.erase();
388 // special case: del root
389 if (p == this->root_)
390 {
391 node_type *q = pal.get_q();
392 this->root_ = q == p ? 0 : q;
393 }
394 }

396 // data members
397 key_type keys_;
398 ring_buffer<
399 node_type,
400 typename allocator_type::template rebind<node_type>::other> trie_;
401};
