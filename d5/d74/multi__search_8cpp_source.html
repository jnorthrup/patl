20static inline unsigned searchInFile(
21 SuffixSet &suffix,
22 const std::string &filename)
23{
24 MatchVector matches;
25 {
26 patl::aux::file_mapping_input fmi(filename.c_str());
27 std::auto_ptr<patl::aux::file_mapping_view> fmv =
28 fmi.create_view(1 << 30);
29 char * const huge = static_cast<char*>(fmv->get_view());
30 const word_t N = fmv->size();
31 const char * const end = huge + N - 1;
32 //
33 SuffixSet::match_iterator mit(&suffix, huge);
34 unsigned line = 1;
35 const char *lineBeg = huge;
36 for (unsigned i = 0; i != N; ++i, ++mit)
37 {
38 const char
39 *pattern = mit->key(),
40 *str = mit.key();
41 //
42 if (*str == '\n')
43 {
44 ++line;
45 lineBeg = str + 1;
46 }
47 // находится ли ключ между двумя '\n'?
48 // т.е., найдено ли полное совпадение с одним из ключей?
49 if ('\n' == pattern[-1] && '\n' == pattern[mit.length()])
50 {
51 const char
52 *stop = "\r\n",
53 *lineEnd = std::find_first_of(
54 str,
55 patl::impl::get_min(str + 160, end),
56 stop,
57 stop + sizeof(stop));
58 std::string lineS(
59 patl::impl::get_max(lineBeg, str - 12),
60 patl::impl::get_min(lineEnd, str + mit.length() + 12));
61 std::replace(lineS.begin(), lineS.end(), '\t', ' ');
62 matches.push_back(std::make_pair(line, lineS));
63 }
64 }
65 }
66 //
67 if (!matches.empty())
68 {
69 printf("*** %s\n", filename.c_str());
70 for (
71 MatchVector::const_iterator cit = matches.begin();
72 cit != matches.end();
73 ++cit)
74 printf("%d\t... %s ...\n",
75 cit->first,
76 cit->second.c_str());
77 printf("\n");
78 }
79 return matches.size();
80}

82static inline unsigned searchByMask(
83 SuffixSet &suffix,
84 const std::string &folder,
85 const std::string &mask)
86{
87 const bool starp = mask == "*" || mask == "*.*";
88 unsigned matc = 0;
89 WIN32_FIND_DATA wfData;
90 HANDLE hFind = FindFirstFile((folder + '\\' + mask).c_str(), &wfData);
91 if (hFind != INVALID_HANDLE_VALUE)
92 {
93 do
94 {
95 if (!(wfData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY))
96 matc += searchInFile(suffix, folder + '\\' + wfData.cFileName);
97 else if (starp && !patl::aux::is_dots_folder(wfData.cFileName))
98 matc += searchByMask(suffix, folder + '\\' + wfData.cFileName, mask);
99 } while (FindNextFile(hFind, &wfData));
100 FindClose(hFind);
101 }
102 if (!starp)
103 {
104 hFind = FindFirstFile((folder + "\\*").c_str(), &wfData);
105 if (hFind != INVALID_HANDLE_VALUE)
106 {
107 do
108 {
109 if ((wfData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) &&
110 !patl::aux::is_dots_folder(wfData.cFileName))
111 matc += searchByMask(
112 suffix,
113 folder + '\\' + wfData.cFileName,
114 mask);
115 } while (FindNextFile(hFind, &wfData));
116 FindClose(hFind);
117 }
118 }
119 return matc;
120}

122static inline unsigned search(SuffixSet &suffix, const std::string &str)
123{
124 const std::string folder = patl::aux::take_file_folder(str);
125 std::string nameExt = patl::aux::take_file_name_ext(str);
126 if (nameExt.empty())
127 nameExt = "*";
128 const bool isWildcard = nameExt.find_first_of("?*") != std::string::npos;
129 if (isWildcard)
130 return searchByMask(suffix, folder, nameExt);
131 WIN32_FIND_DATA wfData;
132 HANDLE hFind = FindFirstFile((folder + '\\' + nameExt).c_str(), &wfData);
133 if (hFind != INVALID_HANDLE_VALUE)
134 {
135 FindClose(hFind);
136 if (wfData.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY)
137 return searchByMask(suffix, str, "*");
138 else
139 return searchInFile(suffix, folder + '\\' + wfData.cFileName);
140 }
141 return 0;
142}
