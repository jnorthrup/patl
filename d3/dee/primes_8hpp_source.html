25template <typename Natural>
26class primes
27 : public generator_base<primes<Natural>, Natural>
28{
29public:
30 typedef Natural nat_type;
31
32 enum { B = 12 << 10 };
33
34 primes(const nat_type &n = 2, const nat_type &m = ~nat_type(0))
35 : n_(n)
36 , m_(get_max<nat_type>(9, m & 1 ? m : m + 1))
37 { }
38
39 PATL_EMIT
40 {
41 if (n_ <= 2) PATL_YIELD(2);
42 if (n_ <= 3) PATL_YIELD(3);
43 if (n_ <= 5) PATL_YIELD(5);
44 if (n_ <= 7) PATL_YIELD(7);
45 for (n_ = get_max<nat_type>(9, n_ & 1 ? n_ : n_ - 1); n_ < m_; )
46 {
47 bound_ = get_min(static_cast<word_t>((m_ - n_) / 2), B * bits_in_word);
48 if (n_ < B)
49 bound_ = get_min(static_cast<word_t>((n_ * n_ - n_) / 2), bound_);
50 //
51 {
52 {
53 const word_t j = static_cast<word_t>((n_ - 3) / 2 % 105);
54 for (word_t i = 0, bmi = offs_[j % bits_in_word] + j / bits_in_word
55 ; i != bound_ / bits_in_word + 1
56 ; ++i, bmi = (bmi + 1) % 105)
57 arr_[i] = bm357_[bmi];
58 arr_[bound_ / bits_in_word] &=
59 ~word_t(0) >> (bits_in_word - bound_ % bits_in_word);
60 }
61 const nat_type sq_m(sqrt_int(n_ + bound_ * 2) + 1);
62 if (sq_m >= 11)
63 {
64 std::auto_ptr<primes> ps(new primes(9, sq_m));
65 if (ps->next())
66 {
67 do
68 {
69 const word_t i = static_cast<word_t>(ps->value());
70 for (word_t j = correct(corr0(static_cast<word_t>(n_ % i), i), i) / 2
71 ; j < bound_
72 ; j += i)
73 arr_[j / bits_in_word] |= 1 << (j % bits_in_word);
74 } while (ps->next());
75 }
76 }
77 }
78 //
79 for (i_ = 0; ; ++i_, n_ += 2)
80 {
81 {
82 word_t
83 sh = i_ % bits_in_word,
84 a = ~arr_[i_ / bits_in_word] >> sh;
85 while (!a)
86 {
87 i_ += bits_in_word - sh;
88 n_ += 2 * (bits_in_word - sh);
89 sh = 0;
90 a = ~arr_[i_ / bits_in_word];
91 }
92 const word_t id = get_lowest_bit_id(a);
93 i_ += id;
94 n_ += 2 * id;
95 }
96 if (i_ >= bound_)
97 break;
98 PATL_YIELD(n_);
99 }
100 }
101 }
102 PATL_STOP_EMIT
103
104private:
105 static word_t corr0(word_t n, word_t i)
106 { return n ? i - n : n; }
107
108 static word_t correct(word_t n, word_t i)
109 { return n & 1 ? n + i : n; }
110
111 nat_type n_, m_;
112 word_t arr_[B + 1];
113 word_t bound_, i_;
114
115 static const word_t offs_[bits_in_word];
116 static const word_t bm357_[105];
117};
