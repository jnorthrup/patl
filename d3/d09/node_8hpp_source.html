131 bool integrity() const

133 const word_t

135 tag0 = get_xtag(0),
136 tag1 = get_xtag(1);
137 const node_type
138 *p0 = get_xlink(0),
139 *p1 = get_xlink(1);
140 return
141 skip == ~word_t(0) && p1 == 0 && get_parent() == 0 && get_parent_id() == 0 ||
142 !tag0 && p0->get_parent() == this && !p0->get_parent_id() && skip < p0->get_skip() ||
143 !tag1 && p1->get_parent() == this && p1->get_parent_id() && skip < p1->get_skip() ||
144 tag0 && tag1;

175 std::swap(tagsid_, old_root->tagsid_);
176#endif
177 old_root->get_parent()->set_xlinktag(old_root->get_parent_id(), old_root, 0);
178 if (!old_root->get_xtag(0))
179 old_root->get_xlink(0)->set_parentid(old_root, 0);
180 if (!old_root->get_xtag(1))
181 old_root->get_xlink(1)->set_parentid(old_root, 1);
182 get_xlink(0)->set_parentid(reinterpret_cast<node_type*>(this), 0);
183 }
