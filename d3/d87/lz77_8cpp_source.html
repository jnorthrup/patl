16template <typename Input, typename Output>
17int lz_compress(unsigned n_pow, unsigned w_pow, unsigned min_match, Input &src, Output &dst)
18{
19 int retcod = 0;
20 //
21 const unsigned
22 n = 1 << n_pow,
23 w = 1 << w_pow;
24 unsigned count = 0;
25 long long total = 0;
26 const char stop_bytes[] =
27 "\x81\xAE\xA4\x11\xB6\x18\x1C\xA7"
28 "\xF5\xE8\xF9\xD2\x7C\x2D\x0E\x3C";
29 char
30 *const buf = new char [2 * n + w + sizeof(stop_bytes)],
31 *const buf2 = buf + n,
32 *const buf3 = buf2 + n;
33 memcpy(buf3 + w, stop_bytes, sizeof(stop_bytes));
34 suffix_t sufs(buf2, n);
35 patl::aux::bit_output<Output> bdst(dst);
36 bdst.put_fib_code(n_pow);
37 //put_fib_code(bdst,w_pow);
38 bdst.put_fib_code(min_match);
39 char *tail;
40 for (bool first_time = true, incomplete = false;
41 ;
42 first_time = false)
43 {
44 if (!incomplete)
45 {
46 char *beg = first_time ? buf2 : buf2 + w;
47 const unsigned reindex_cnt = first_time ? 0 : sufs.count_reindex(beg);
48 const bool need_reindex = reindex_cnt < sufs.size() / 4;
49 if (!first_time && need_reindex)
50 sufs.pop_reindex(reindex_cnt);
51 const unsigned
52 toread = first_time ? n + w : n,
53 avail = src(beg, toread);
54 if (!first_time)
55 {
56 if (need_reindex)
57 sufs.push_reindex(reindex_cnt);
58 else
59 {
60 unsigned size = sufs.size();
61 sufs.clear();
62 sufs.push_reindex(size);
63 }
64 }
65 if (!avail)
66 break;
67 incomplete = avail < toread;
68 tail = beg + std::min(toread, avail);
69 }
70 bool copy_state = false;
71 unsigned
72 copy_count = 0,
73 skip = 0;
74 const char
75 *cur = buf2,
76 *end = std::min(buf3, tail);
77 for (;
78 cur != end;
79 ++cur, skip = patl::impl::max0(skip - 1))
80 {
81 if (!first_time)
82 sufs.pop_front();
83 vertex v = sufs.push_back();
84 if (!skip)
85 {
86 skip = std::min(
87 w,
88 std::min(
89 patl::impl::max0(v.skip()) / 8,
90 patl::impl::unsigned_cast(end - cur)));
91 const char *match = v.sibling().key();
92 ++count;
93 if (skip < min_match)
94 {
95 ++total;
96 if (copy_state)
97 {
98 if (copy_count == 65536)
99 {
100 bdst.put_fib_code(1);
101 bdst.put_fib_code(copy_count);
102 bdst.put_bytes(cur - copy_count, copy_count);
103 copy_count = 1;
104 }
105 else
106 ++copy_count;
107 }
108 else
109 {
110 copy_count = 1;
111 copy_state = true;
112 }
113 skip = 0;
114 }
115 else
116 {
117 total += skip;
118 if (copy_state)
119 {
120 bdst.put_fib_code(1);
121 bdst.put_fib_code(copy_count);
122 bdst.put_bytes(cur - copy_count, copy_count);
123 copy_state = false;
124 }
125 bdst.put_fib_code(skip - min_match + 3);
126 bdst.put_bits(cur - match, patl::impl::get_highest_bit_id(sufs.size() - 1) + 1);
127 }
128 }
129 }
130 if (copy_state)
131 {
132 bdst.put_fib_code(1);
133 bdst.put_fib_code(copy_count);
134 bdst.put_bytes(cur - copy_count, copy_count);
135 }
136 if (incomplete && tail < buf3)
137 break;
138 memmove(buf, buf2, n + w + sizeof(stop_bytes));
139 tail -= n;
140 sufs.rebind(buf);
141 }
142 bdst.put_fib_code(2); // EOF
143 delete[] buf;
144 //
145 return retcod;
146}

157template <typename Input, typename Output>
158int lz_decompress(Input &src, Output &dst)
159{
160 int retcod = 0;
161 //
162 patl::aux::bit_input<Input> bsrc(src);
163 const unsigned
164 n_pow = bsrc.get_fib_code(),
165 n = 1 << n_pow,
166 min_match = bsrc.get_fib_code();
167 char
168 *const buf = new char [2 * n/*+w*/],
169 *const buf2 = buf + n,
170 *const buf3 = buf2 + n;
171 for (bool first_time = true; ; first_time = false)
172 {
173 char
174 *cur = buf2,
175 *end = buf3;
176 for (; cur != end; )
177 {
178 const unsigned code = bsrc.get_fib_code();
179 if (code == 1)
180 {
181 const unsigned copy_count = bsrc.get_fib_code();
182 bsrc.get_bytes(cur, copy_count);
183 cur += copy_count;
184 }
185 else if (code == 2)
186 {
187 dst(buf2, cur - buf2);
188 goto eof;
189 }
190 else
191 {
192 const unsigned
193 skip = code - 3 + min_match,
194 match = bsrc.get_bits(patl::impl::get_highest_bit_id(
195 first_time ? cur - buf2 : n - 1) + 1);
196 copymem(cur, cur - match, skip);
197 cur += skip;
198 }
199 }
200 dst(buf2, cur - buf2);
201 memcpy(buf, buf2, n);
202 }
203eof:
204 delete[] buf;
205 //
206 return retcod;
207}
