91template <

95 typename Pred = std::equal_to<Key>,

97class unordered_map
98{
99 typedef Datum mapped_type;
100
101public:
102 typedef Key key_type;
103 typedef std::pair<const key_type, mapped_type> value_type;
104 typedef impl::hash_node<key_type, value_type> node_type;

110 explicit unordered_map(
111 word_t capacity = 0, // min buckets count
112 const hasher &hf = hasher(),
113 const key_equal &eql = key_equal(),
114 const allocator_type &alloc = allocator_type())
115 : hf_(hf)
116 , eql_(eql)
117 , alloc_(alloc)
118 , size_(0)
119 , free_size_(0)
120 {
121 init(capacity);
122 }
123
124 bool empty() const
125 {
126 return !size();
127 }
128
129 word_t size() const
130 {
131 return size_ - free_size_;
132 }
133
134 void clear()
135 {
136 if (size_)
137 {
138 std::fill(buckets_.begin(), buckets_.end(), ~word_t(0));
139 std::fill(nodes_.begin(), nodes_.end(), node_type()); // ???
140 free_list_ = ~word_t(0);
141 free_size_ = size_ = 0;
142 }
143 }
144
145 mapped_type &operator[](const key_type &key)
146 {
147 if (buckets_.empty())
148 init(0);
149 const word_t num = hf_(key) & 0x7FFFFFFF;
150 word_t index = num % buckets_.size();
151 for (word_t i = buckets_[index]; i != ~word_t(0); i = nodes_[i].get_next())
152 {
153 if (nodes_[i].get_hash() == num && eql_(nodes_[i].get_key(), key))
154 return nodes_[i].get_value().second;
155 }
156 word_t free_list;
157 if (free_size_ > 0)
158 {
159 free_list = free_list_;
160 free_list_ = nodes_[free_list].get_next();
161 --free_size_;
162 }
163 else
164 {
165 if (size_ == nodes_.size())
166 {
167 resize();
168 index = num % buckets_.size();
169 }
170 free_list = size_++;
171 }
172 nodes_[free_list].set_all(num, buckets_[index], std::make_pair(key, mapped_type()));
173 return nodes_[free_list].get_value().second;
174 }
175
176 size_type erase(const key_type &key)
177 {
178 if (!buckets_.empty())
179 {
180 const word_t num = hf_(key) & 0x7FFFFFFF;
181 word_t index = num % buckets_.size();
182 word_t num3 = ~word_t(0);
183 for (word_t i = buckets_[index]; i != ~word_t(0); i = nodes_[i].get_next())
184 {
185 if (nodes_[i].get_hash() == num && eql_(nodes_[i].get_key(), key))
186 {
187 if (num3 == ~word_t(0))
188 buckets_[index] = nodes_[i].get_next();
189 else
190 nodes_[num3].set_next(nodes_[i].get_next());
191 nodes_[i].set_all(~word_t(0), free_list_, value_type());
192 free_list_ = i;
193 ++free_size_;
194 return 1;
195 }
196 num3 = i;
197 }
198 }
199 return 0;
200 }

225 void resize()
226 {
227 impl::primes<word_t> ps(size_ * 2);
228 ps.next();
229 const word_t prime = ps.value();
230 std::vector<word_t> num_arr(prime, ~word_t(0));
231 std::vector<node_type> dest_arr(prime);
232 std::copy(nodes_.begin(), nodes_.begin() + size_, dest_arr);
233 for (word_t j = 0; j != size_; ++j)
234 {
235 const word_t index = dest_arr[j].get_hash() % prime;
236 dest_arr[j].set_next(num_arr[index]);
237 num_arr[index] = j;
238 }
239 std::swap(buckets_, num_arr);
240 std::swap(nodes_, dest_arr);
241 }
