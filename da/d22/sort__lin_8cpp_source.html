67template <typename T>
68class bit_comparator<piece<T> >
69 : public bit_comparator<T>
70{
71 typedef bit_comparator<T> super;
72 typedef piece<T> value_type;
73
74public:
75 typedef T char_type;
76
77 static const word_t bit_size = 8 * sizeof(T);
78
79 word_t bit_length(const value_type &s) const
80 {
81 return bit_size * s.length();
82 }
83
84 word_t get_bit(const value_type &s, word_t id) const
85 {
86 return super::get_bit(s[id / bit_size], id % bit_size);
87 }
88
89 word_t bit_mismatch(const value_type &a, const value_type &b) const
90 {
91 if (&a == &b)
92 return ~word_t(0);
93 return b.length() < a.length()
94 ? mismatch_intern(b, a)
95 : mismatch_intern(a, b);
96 }
97
98private:
99 word_t mismatch_intern(const value_type &a, const value_type &b) const
100 {
101 word_t i = 0;
102 for (; i != a.length(); ++i)
103 {
104 const word_t m = super::bit_mismatch(a[i], b[i]);
105 if (m != ~word_t(0))
106 return i * bit_size + m;
107 }
108 return ~word_t(0);
109 }
110};

122class strings_from_file_generator
123 : public patl::impl::generator_base<strings_from_file_generator, line_t>
124{
125public:
126 strings_from_file_generator(
127 const char *infname,
128 word_t block_size,
129 word_t max_string_size)
130 : fmi_(infname)
131 , block_size_(block_size)
132 , max_string_size_(max_string_size)
133 , lin_num_(0)
134 , skip_yield_(false)
135 , viewed_off_beg_(0)
136 , viewed_off_cur_(0)

138 }
139
140 PATL_EMIT
141 {
142 while (fmi_.residue_length())
143 {
144 fmv_ = fmi_.create_view(block_size_);
145 text_ = static_cast<unsigned char*>(fmv_->get_view());
146 text_end_ = text_ + fmv_->size();
147 lin_beg_ = text_ + viewed_off_beg_;
148 lin_cur_ = text_ + viewed_off_cur_;
149 viewed_off_cur_ = viewed_off_beg_ = 0;
150 for (; lin_cur_ != text_end_; ++lin_cur_)
151 {
152 // проход по тексту, ищем '\n'
153 if (*lin_cur_ == '\n')
154 {
155 ++lin_cur_; // указывает на следующий после '\n' символ
156 ++lin_num_;
157 // yield, если длина строки не больше допустимой
158 if (static_cast<word_t>(lin_cur_ - lin_beg_) <= max_string_size_ &&
159 !skip_yield_)
160 PATL_YIELD(line_t(lin_beg_, lin_cur_));
161 else
162 printf("ignore too big line number %d\n", lin_num_);
163 skip_yield_ = false;
164 // находим начало следующей строки, пропускаем блоки '\n'
165 for (lin_beg_ = lin_cur_
166 ; lin_beg_ != text_end_ && *lin_beg_ == '\n'
167 ; ++lin_beg_) ;
168 lin_cur_ = lin_beg_;
169 if (lin_beg_ == text_end_)
170 break;
171 }
172 }
173 if (lin_beg_ != text_end_ && !fmi_.residue_length() &&
174 static_cast<word_t>(lin_cur_ - lin_beg_) <= max_string_size_ &&
175 !skip_yield_) // yield последней строки
176 {
177 ++lin_num_;
178 PATL_YIELD(line_t(lin_beg_, lin_cur_));
179 }
180 PATL_YIELD(line_t()); // условный сигнал - окно закрывается
181 if (!fmi_.residue_length())
182 break;
183 // если строка не завершена - откатываем окно назад, до начала строки
184 if (lin_beg_ != text_end_)
185 {
186 if (static_cast<word_t>(lin_cur_ - lin_beg_) > max_string_size_)
187 skip_yield_ = true; // строка слишком большая, пропустить остаток
188 else
189 {
190 viewed_off_cur_ = fmi_.drop_back(lin_cur_ - lin_beg_);
191 viewed_off_beg_ = viewed_off_cur_ - (lin_cur_ - lin_beg_);
192 }
193 }
194 }
195 }
196 PATL_STOP_EMIT

324 {

329 reinterpret_cast<strings_from_file_generator*>(0)));

331 }
332 else

341 printf("\r% 3u%% => 0%% of sort completed",

343 }
344 if (merge_src_.size() == 1) // все поместилось в один файл
345 break;
346 else if (first_run_)
347 first_run_ = false;
348 // заполняем первые D строк, насколько хватит входных временных файлов
349 for (it_ = merge_src_.begin()
350 ; it_ != merge_src_.end() && trie_.size() != D_

354 it_->first.c_str(),
355 max_string_size_,
356 max_string_size_);

361#endif
362 inserted_ = false;
363 do
364 {
365 strings_continued_ = it_->second->next();
366 if (strings_continued_)
367 inserted_ =
368 !it_->second->value().empty() &&
369 trie_.insert(std::make_pair(it_->second->value(), it_)).second;
370 } while (!inserted_ && strings_continued_);
371 // нельзя удалять генератор строк из файла до окончания использования этих строк
372 if (!strings_continued_ && !inserted_)
373 delete_src();
374 else
375 ++it_;
376 }
377 while (trie_.size() > 1)
378 {
379 itt_ = trie_.begin();
380 PATL_YIELD(itt_->first);
381 it_ = itt_->second;
382 trie_.erase(itt_);
383 if (!(it_->second->get_lin_num() & 0xFFFF))
384 {
385 const long long length = it_->second->get_file_length();
386 printf("\r% 3u%% => % 3u%% of sort completed",
387 amount_per100_,
388 static_cast<word_t>(100 * (length - it_->second->residue_length()) / length));
389 }
390 inserted_ = false;
391 strings_continued_ = !it_->second->empty();
392 do
393 {
394 strings_continued_ = strings_continued_ && it_->second->next();
395 if (strings_continued_)
396 inserted_ =
397 !it_->second->value().empty() &&
398 trie_.insert(std::make_pair(it_->second->value(), it_)).second;
399 } while (!inserted_ && strings_continued_);
400 if (!strings_continued_ && !inserted_)
401 delete_src();
402 }
403 // остался только один источник, берем строки сразу из него
404 if (trie_.size() == 1)
405 {
406 itt_ = trie_.begin();
407 PATL_YIELD(itt_->first);
408 it_ = itt_->second;
409 trie_.clear();
410 for (strings_continued_ = !it_->second->empty()
411 ; strings_continued_ && it_->second->next()
412 ; )
413 if (!it_->second->value().empty())
414 PATL_YIELD(it_->second->value());
415 delete_src();
416 }
417 }
418 printf("\r100%% => 100%% of sort completed\n");
419 }

458void sort_huge_text_file(const char *infname, const char *outfname, word_t N)
459{
460 const word_t D = 4; // block's amount
461 const word_t max_string_size = patl::impl::align_up<1 << 16>((N << 19) / D);
462 std::string tmpfname(patl::aux::get_temp_file_name());
463 std::auto_ptr<patl::aux::buffered_output> outbuf(
464 new patl::aux::buffered_output(tmpfname.c_str()));
465 strings_from_merged_files_generator sfmf_gen(infname, N, D, max_string_size);
466 while (sfmf_gen.next())
467 {
468 const line_t value(sfmf_gen.value());
469 if (value.empty())
470 {
471 sfmf_gen.push_front(merge_src_t::value_type(
472 tmpfname,
473 reinterpret_cast<strings_from_file_generator*>(0)));
474#ifdef SORT_LIN_DEBUG
475 outbuf->flush_buffer();
476 printf("\nmerged to file: '%s', %5.3g Mb ***\n",
477 tmpfname.c_str(),
478 static_cast<double>(outbuf->get_file_length()) / 1048576.0);
479#endif
480 tmpfname = patl::aux::get_temp_file_name();
481 outbuf.reset(new patl::aux::buffered_output(tmpfname.c_str()));
482 }
483 else
484 outbuf->write_bytes(value.begin(), value.length());
485 }
486 outbuf.reset();
487 if (sfmf_gen.sole_tmpfname())
488 {
489 patl::aux::delete_file(tmpfname.c_str());
490 patl::aux::move_file(sfmf_gen.first_tmpfname().c_str(), outfname);
491 }
492 else
493 patl::aux::move_file(tmpfname.c_str(), outfname);
494}
