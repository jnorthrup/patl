106 void next_fit()
107 {
108 for (; ; ++pit_)
109 for (;;)
110 {
111 while (pit_->get_qtag()) // reach the leaf
112 {
113 if (pit_->the_end() || prove_branch(true))
114 return;
115 }
116 if (prove_branch(false))
117 break;
118 }
119 }
120
121 bool prove_branch(bool leaf)
122 {
123 const word_t skip0 = max0(static_cast<sword_t>(pit_->skip()));
124 // if current bit is illegal in pattern - skip whole subtree
125 if (!decis_.bit_level(skip0, pit_->get_qid()))
126 {
127 pit_.next_subtree();
128 return false;
129 }
130 word_t i = skip0 / bit_compare::bit_size;
131 const key_type &key = pit_->key();
132 const word_t limit =
133 (leaf ? pit_->bit_comp().bit_length(key) : pit_->next_skip()) /
134 bit_compare::bit_size;
135 for (; i != limit; ++i)
136 {
137 if (!decis_(i))
138 {
139#if defined(_MSC_VER)
140# pragma warning(push)
141# pragma warning(disable : 4127) // C4127: conditional expression is constant
142#endif
143 if (Decision::accept_subtree)
144 return true;
145#if defined(_MSC_VER)
146# pragma warning(pop)
147#endif
148 pit_.next_subtree();
149 return false;
150 }
151 if (!decis_(i, key[i]))
152 {
153 pit_.next_subtree();
154 return false;
155 }
156 }
157 return true;
158 }

160 Decision decis_;

162protected:
163 preorder_iterator pit_;
