66void multiple_common_substring(
67 const std::vector<std::string> &strs,
68 std::vector<std::string> &align)
69{
70 if (strs.size() == 1)
71 {
72 align.push_back(strs[0]);
73 align.push_back(std::string(strs[0].length(), ' '));
74 return;
75 }
76 //
77 std::vector<char> seq;
78 std::vector<id_type> ids;
79 std::vector<word_t> offs;
80 for (id_type id = 0; id != strs.size(); ++id)
81 {
82 offs.push_back(seq.size());
83 std::copy(strs[id].begin(), strs[id].end(), std::back_inserter(seq));
84 seq.push_back('\0');
85 seq.push_back(id);
86 ids.insert(ids.end(), strs[id].length() + 2, id);
87 }
88 offs.push_back(seq.size());
89 //
90 const word_t K = strs.size();
91 //
92 suffix_t suf(&seq[0], seq.size());
93 for (word_t i = 0; i != seq.size() - 1; ++i)
94 suf.push_back();
95 //
96 std::vector<word_t> h(suf.size() * 2);
97 //
98 {
99 patl::lca_oracle<suffix_t> lca(suf);
100 //
101 std::vector<const_vertex> last_vtx(K);
102 //
103 for (suffix_t::const_iterator it = suf.begin(); it != suf.end(); ++it)
104 {
105 const word_t k = ids[*it - suf.keys()];
106 const const_vertex &vtx = static_cast<const const_vertex&>(it);
107 if (last_vtx[k].compact())
108 ++h[suf.vertex_index_of(lca(last_vtx[k], vtx))];
109 last_vtx[k] = vtx;
110 }
111 }
112 //
113#ifdef MULTI_ALIGN_DEBUG
114 {
115 std::ofstream fout("malign_suf.dot");
116 patricia_dot_base<suffix_t> pdb(&ids, &h);
117 patl::patricia_dot_creator<
118 suffix_t,
119 std::ofstream, patricia_dot_base<suffix_t> > dotcr(fout, pdb);
120 dotcr.create(suf.root());
121 }
122#endif
123 //
124 std::vector<const_vertex> v(K + 1);
125 std::vector<word_t> v_len(K + 1);
126 //
127 std::vector<word_t> s, u;
128 const const_vertex root = suf.root();
129 const suffix_t::const_postorder_iterator
130 pit_beg = root.postorder_begin(),
131 pit_end = root.postorder_end();
132 for (suffix_t::const_postorder_iterator pit = pit_beg; pit != pit_end; ++pit)
133 {
134 if (pit->get_qtag())
135 {
136 s.push_back(1);
137 u.push_back(0);
138 }
139 else
140 {
141 const word_t
142 s_v = s.back() + s[s.size() - 2],
143 u_v = u.back() + u[u.size() - 2] + h[suf.vertex_index_of(*pit)],
144 c_v = s_v - u_v,
145 idx = suf.index_of(*pit),
146 len = patl::impl::get_min(
147 pit->next_skip() / 8,
148 patl::impl::max0(strs[ids[idx]].length() - (idx - offs[ids[idx]])));
149 if (len > v_len[c_v])
150 {
151 v[c_v] = *pit;
152 v_len[c_v] = len;
153 }
154 s.pop_back();
155 s.back() = s_v;
156 u.pop_back();
157 u.back() = u_v;
158 }
159 }
160 //
161 for (word_t k = K - 1; k != 1; --k)
162 {
163 if (v_len[k] < v_len[k + 1])
164 {
165 v[k] = v[k + 1];
166 v_len[k] = v_len[k + 1];
167 }
168 }
169 // multiple common substrings
170#ifdef MULTI_ALIGN_DEBUG
171 printf("---\n");
172 for (word_t k = 2; k != v.size(); ++k)
173 {
174 if (v[k].compact())
175 printf("%u\t%u\t%s\n", k, v_len[k], std::string(v[k].key(), v_len[k]).c_str());
176 }
177#endif
178 // эвристический выбор подход€щего k
179 word_t k_cur = v.size() - 1;
180 for (; k_cur != 1 && v_len[k_cur] < 2; --k_cur) ;
181 if (k_cur == 1)
182 {
183 for (k_cur = v.size() - 1; k_cur != 1 && !v_len[k_cur]; --k_cur) ;
184 if (k_cur == 1)
185 {
186 word_t max_len = 0;
187 for (id_type i = 0; i != strs.size(); ++i)
188 {
189 if (strs[i].length() > max_len)
190 max_len = strs[i].length();
191 }
192 for (id_type i = 0; i != strs.size(); ++i)
193 {
194 align.push_back(strs[i]);
195 std::string &ref = align.back();
196 while (ref.length() != max_len)
197 ref.push_back(' ');
198 }
199 align.push_back(std::string(max_len, ' '));
200 return;
201 }
202 }
203 //
204 // std::pair<номер строки, смещение в строке>
205 typedef std::map<id_type, word_t> motif_map;
206 motif_map motifs; // номера строк с повтор€ющимс€ мотивом
207 {
208 const suffix_t::const_iterator it_end = v[k_cur].end();
209 for (suffix_t::const_iterator it = v[k_cur].begin(); it != it_end; ++it)
210 {
211 const id_type id = ids[*it - suf.keys()];
212 const word_t off = *it - suf.keys() - offs[id];
213 motifs[id] = off;
214 }
215 }
216 const word_t motif_len = v_len[k_cur];
217 const std::string
218 motif(v[k_cur].key(), motif_len),
219 padding(motif_len, ' ');
220 align.clear();
221 for (id_type i = 0; i != K; ++i)
222 align.push_back(motifs.find(i) != motifs.end() ? motif : padding);
223 align.push_back(std::string(motif_len, '^'));
224 word_t
225 sum_left = 0,
226 sum_right = 0;
227 for (motif_map::const_iterator it = motifs.begin(); it != motifs.end(); ++it)
228 {
229 if (it->second)
230 ++sum_left;
231 if (strs[it->first].length() - it->second - motif_len)
232 ++sum_right;
233 }
234 // left side
235 if (sum_left)
236 {
237 std::vector<id_type> left_ids;
238 std::vector<std::string> left_strs;
239 for (id_type i = 0; i != K; ++i)
240 {
241 const motif_map::const_iterator it = motifs.find(i);
242 if (it != motifs.end())
243 {
244 if (it->second)
245 {
246 left_ids.push_back(it->first);
247 left_strs.push_back(strs[it->first].substr(0, it->second));
248 }
249 }
250 else
251 {
252 left_ids.push_back(i);
253 left_strs.push_back(strs[i]);
254 }
255 }
256 std::vector<std::string> left_align;
257 multiple_common_substring(left_strs, left_align);
258 const word_t left_len = left_align[0].length();
259 const std::string left_pad(left_len, ' ');
260 for (id_type i = 0, j = 0; i != K; ++i)
261 {
262 align[i].insert(0, j != left_ids.size() && left_ids[j] == i
263 ? left_align[j++]
264 : left_pad);
265 }
266 align.back().insert(0, left_align.back());
267 }
268 // right side
269 if (sum_right)
270 {
271 std::vector<id_type> right_ids;
272 std::vector<std::string> right_strs;
273 for (id_type i = 0; i != K; ++i)
274 {
275 const motif_map::const_iterator it = motifs.find(i);
276 if (it != motifs.end())
277 {
278 if (strs[it->first].length() - it->second - motif_len)
279 {
280 right_ids.push_back(it->first);
281 right_strs.push_back(strs[it->first].substr(it->second + motif_len));
282 }
283 }
284 else if (!sum_left)
285 {
286 right_ids.push_back(i);
287 right_strs.push_back(strs[i]);
288 }
289 }
290 std::vector<std::string> right_align;
291 multiple_common_substring(right_strs, right_align);
292 const word_t right_len = right_align[0].length();
293 const std::string right_pad(right_len, ' ');
294 for (id_type i = 0, j = 0; i != K; ++i)
295 {
296 align[i].append(j != right_ids.size() && right_ids[j] == i
297 ? right_align[j++]
298 : right_pad);
299 }
300 align.back().append(right_align.back());
301 }
302 // neither left nor right
303 if (!sum_left && !sum_right && K - k_cur)
304 {
305 std::vector<id_type> neither_ids;
306 std::vector<std::string> neither_strs;
307 for (id_type i = 0; i != K; ++i)
308 {
309 const motif_map::const_iterator it = motifs.find(i);
310 if (it == motifs.end())
311 {
312 neither_ids.push_back(i);
313 neither_strs.push_back(strs[i]);
314 }
315 }
316 std::vector<std::string> neither_align;
317 multiple_common_substring(neither_strs, neither_align);
318 const word_t neither_len = neither_align[0].length();
319 const std::string neither_pad(neither_len, ' ');
320 for (id_type i = 0, j = 0; i != K; ++i)
321 {
322 align[i].append(j != neither_ids.size() && neither_ids[j] == i
323 ? neither_align[j++]
324 : neither_pad);
325 }
326 align.back().append(neither_align.back());
327 }
328}
