108 {
109 // pop_back-while removed to determ(...)
110 if (i < super::mask_len_ && ch != super::terminator_)
111 {
112 if (super::mask_[i] == ch)
113 return true;
114 if (diff_.size() == dist_)
115 return false;
116 diff_.push_back(i);
117 return true;
118 }
119 return (!SameLength && i >= super::mask_len_) ||
120 (i == super::mask_len_ && ch == super::terminator_);
121 }
122
124 bool bit_level(word_t skip, word_t id)
125 {
126 while (!diff_.empty() &&
127 skip / super::bit_compare::bit_size <= diff_.back())
128 diff_.pop_back();
129 return diff_.size() == dist_ && skip < super::mask_bit_len_
130 ? super::bit_comp_.get_bit(super::mask_, skip) == id
131 : true;
132 }
133
134private:
135 word_t dist_;
136 std::vector<word_t> diff_;
