63 void init()
64 {
65 // deallocate old
66 lca_alloc_.deallocate(lca_, lca_size_);
67 lca_ = 0;
68 unsigned_alloc_.deallocate(lca_map_, lca_size_ + 1);
69 lca_map_ = 0;
70 //
71 if (cont_->size() == 0)
72 return;
73 lca_ = lca_alloc_.allocate(lca_size_ = 2 * cont_->size());
74 memset(lca_, 0, lca_size_ * sizeof(lca_type));
75 lca_map_ = unsigned_alloc_.allocate(lca_size_ + 1);
76 // numerate, set I- & L-nodes
77 word_t num = 0;
78 vertex vtx(cont_->root());
79 const vertex vtxEnd(vtx.sibling());
80 while (vtx != vtxEnd)
81 {
82 // descend to the leaves
83 for (; !vtx.get_qtag(); vtx.iterate(0))
84 get_by(vtx)->numerate(++num);
85 //
86 lca_type *cur = get_by(vtx);
87 cur->numerate(++num);
88 cur->set_inode();
89 // move to the next
90 if (vtx.get_qid())
91 {
92 while (vtx.get_parent_id())
93 {
94 vtx.ascend();
95 get_by(vtx)->setup_i(this);
96 }
97 //
98 vtx.ascend();
99 get_by(vtx)->setup_i(this);
100 }
101 vtx.toggle();
102 }
103 // num - last numerator
104 lca_root_h_ = impl::get_highest_bit_id(num);
105 // set A bits
106 vtx = cont_->root();
107 lca_type *cur = get_by(vtx);
108 cur->set_root_a();
109 if (!vtx.get_qtag())
110 vtx.iterate(0);
111 for (; vtx != vtxEnd; vtx.template move_subtree<1>())
112 {
113 // descend to the leaves
114 for (; !vtx.get_qtag(); vtx.iterate(0))
115 get_by(vtx)->setup_a(this);
116 //
117 get_by(vtx)->setup_a(this);
118 }
119 // generate map: number -> index
120 for (word_t i = 0; i != lca_size_; ++i)
121 lca_map_[lca_[i].number()] = i;
122 }

125 vertex operator()(vertex vx, vertex vy) const
126 {
127 const lca_type
128 *xPrime = get_by(vx),
129 *xI = xPrime->get_i(),
130 *yPrime = get_by(vy),
131 *yI = yPrime->get_i();
132 const word_t
133 b = impl::get_binary_lca(lca_root_h_, xI->number(), yI->number()),
134 i = impl::get_lowest_bit_id(b),
135 xA = xPrime->get_a(),
136 yA = yPrime->get_a(),
137 j = impl::get_lowest_bit_id((xA & yA) >> i << i);
138 //
139 vertex
140 xCap = new_algo(xPrime),
141 yCap = new_algo(yPrime);
142 if (impl::get_lowest_bit_id(xA) < j)
143 {
144 const word_t
145 k = impl::get_highest_bit_id(xA & (word_t(1) << j) - 1),
146 wINum = (xI->number() >> k | word_t(1)) << k;
147 const lca_type
148 *wI = lca_ + lca_map_[wINum],
149 *wLI = wI->get_l_i();
150 xCap = new_algo(wLI);
151 xCap.ascend();
152 }
153 if (impl::get_lowest_bit_id(yA) < j)
154 {
155 const word_t
156 k = impl::get_highest_bit_id(yA & (word_t(1) << j) - 1),
157 wINum = (yI->number() >> k | word_t(1)) << k;
158 const lca_type
159 *wI = lca_ + lca_map_[wINum],
160 *wLI = wI->get_l_i();
161 yCap = new_algo(wLI);
162 yCap.ascend();
163 }
164 return get_by(xCap)->number() < get_by(yCap)->number() ? xCap : yCap;
165 }
