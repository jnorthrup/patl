118
119 bool is_data() const
120 {
121 return cur_ != end_;
122 }
123
124 unsigned char read_byte()
125 {
126 const unsigned char byte = *cur_++;
127 if (cur_ == end_)
128 fill_buffer();
129 return byte;
130 }
131
132 word_t read_word()
133 {
134 word_t w = static_cast<word_t>(read_byte());
135 w |= static_cast<word_t>(read_byte()) << 8;
136 w |= static_cast<word_t>(read_byte()) << 16;
137 w |= static_cast<word_t>(read_byte()) << 24;
138#ifdef PATL_64
139 w |= static_cast<word_t>(read_byte()) << 32;
140 w |= static_cast<word_t>(read_byte()) << 40;
141 w |= static_cast<word_t>(read_byte()) << 48;
142 w |= static_cast<word_t>(read_byte()) << 56;
143#endif
144 return w;
145 }
146
147 word_t read_bytes(unsigned char *ptr, word_t len)
148 {
149 const word_t
150 avail = static_cast<word_t>(end_ - cur_),
151 chunk = avail < len ? avail : len;
152 std::copy(cur_, cur_ + chunk, ptr);
153 cur_ += chunk;
154 len -= chunk;
155 if (len)
156 len = fill_buffer(ptr + chunk, len);
157 if (cur_ == end_)
158 fill_buffer();
159 return chunk + len;
160 }
161
162 void drop_back(word_t offs)
163 {
164 if (static_cast<word_t>(cur_ - buf_) < offs)
165 {
166 set_file_ptr(get_file_ptr() - (end_ - cur_) - offs);
167 fill_buffer();
168 }
169 else
170 cur_ -= offs;
171 }
172
173 void skip_data(word_t offs)
174 {
175 if (static_cast<word_t>(end_ - cur_) < offs)
176 {
177 set_file_ptr(get_file_ptr() - (end_ - cur_) + offs);
178 fill_buffer();
179 }
180 else
181 cur_ += offs;
182 }
183
184private:
185 void fill_buffer()
186 {
187 end_ = (cur_ = buf_) + fill_buffer(buf_, BufferSize);
188 }
189
190 word_t fill_buffer(unsigned char *ptr, word_t len)
191 {
192 DWORD realSize;
193 const BOOL ret = ::ReadFile(fh_, ptr, len, &realSize, 0);
194 if (!ret)
195 throw std::exception();
196 return realSize;
197 }

199 unsigned char *cur_;
200};

223 *end_++ = byte;

227
228 void write_word(word_t dw)
229 {
230 const unsigned char *b =
231 reinterpret_cast<unsigned char*>(&dw);
232 write_byte(b[0]);
233 write_byte(b[1]);
234 write_byte(b[2]);
235 write_byte(b[3]);
236#ifdef PATL_64
237 write_byte(b[4]);

239 write_byte(b[6]);
240 write_byte(b[7]);
241#endif
242 }
243
244 void write_string(const char *str)
245 {
246 for (; *str != 0; ++str)
247 write_byte(*str);
248 }
249
250 void write_bytes(const unsigned char *ptr, word_t len)
251 {
252 const word_t
253 avail = static_cast<word_t>(buf_ + BufferSize - end_),
254 chunk = avail < len ? avail : len;
255 std::copy(ptr, ptr + chunk, end_);
256 end_ += chunk;
257 len -= chunk;
258 if (end_ == buf_ + BufferSize)
259 flush_buffer();
260 if (len)
261 flush_buffer(ptr + chunk, len);
262 }
263
264 void move_offset(sword_t offs)
265 {
266 flush_buffer();
267 set_file_ptr(get_file_ptr() + offs);
268 }
269
270 void flush_buffer()
271 {
272 flush_buffer(buf_, end_ - buf_);
273 end_ = buf_;
274 }
275
276private:
277 void flush_buffer(const unsigned char *ptr, word_t len)
278 {
279 DWORD realSize;
280 const BOOL ret = ::WriteFile(fh_, ptr, len, &realSize, 0);
281 if (!ret || realSize != len)
282 throw std::exception();
283 }
