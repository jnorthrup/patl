20template <typename This>
21class range_coder_base
22{
23 typedef This this_t;
24
25public:
26 enum { BOT = 1 << 16, TOP = 1 << 24 }; // currently 32-bit only
27
28 range_coder()
29 : low_(0)
30 , range_(~word_t(0))
31 {
32 }
33
34private:
35 void normalize()
36 {
37 while (
38 (low_ ^ low_ + range_) < TOP ||
39 range_ < BOT && ((range_ = (0 - low_) & BOT - 1), 1))
40 {
41 inner_normalize();
42 range_ <<= 8;
43 low_ <<= 8;
44 }
45 }

52template <typename Out>
53class range_encoder

56public:

65 {

69 }
70
71 void encode(word_t cum_freq, word_t freq, word_t tot_freq)
72 {
73 PATL_ASSERT(cum_freq + freq <= tot_freq && freq && tot_freq <= BOT);
74 range_ /= tot_freq;
75 low_ += cum_freq * range_;
