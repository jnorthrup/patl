55 void create(const const_vertex &vtx, bool clustering = false) const
56 {
57 out_ << "\n// BEGIN create from vertex\n";
58 out_ << "\n// node definitions (preorder depth-first)\n";
59 const word_t
60 init_id = vtx.get_qid(),
61 sibl_qtag = vtx.sibling().get_qtag();
62 out_ << "sibling[shape = plaintext, label = "
63 << (vtx.skip() == ~word_t(0) ? "\"nil\"" : "\"...\"")
64 << "]\n";
65 const const_preorder_iterator pit_end(vtx.preorder_end());
66 for (const_preorder_iterator pit(vtx.preorder_begin()); pit != pit_end; ++pit)
67 if (init_id == pit->get_qid())
68 out_node(*pit);
69 //
70 out_ << std::hex;
71 //
72 out_ << "\n// root link to sibling\n";
73 out_ << 'n' << vtx.node_q_uid() << "->sibling[tailport = "
74 << (init_id ? "sw" : "se") << ", headport = "
75 << (sibl_qtag ? (init_id ? "se" : "sw") : (init_id ? "ne" : "nw")) << ", style = "
76 << (sibl_qtag ? "dotted" : "solid") << "]\n";
77 out_ << "\n// forward left links\n";
78 out_ << "edge[tailport = sw, weight = 2]\n";
79 for (const_preorder_iterator pit(vtx.preorder_begin()); pit != pit_end; ++pit)
80 {
81 if (!pit->get_qtag() && !pit->get_qid())
82 out_edge(*pit);
83 }
84 out_ << "\n// forward right links\n";
85 out_ << "edge[tailport = se]\n";
86 for (const_preorder_iterator pit(vtx.preorder_begin()); pit != pit_end; ++pit)
87 {
88 if (!pit->get_qtag() && pit->get_qid())
89 out_edge(*pit);
90 }
91 out_ << "\n// backward left links\n";
92 out_ << "edge[style = dotted, tailport = sw, weight = 1]\n";
93 for (const_preorder_iterator pit(vtx.preorder_begin()); pit != pit_end; ++pit)
94 {
95 if (pit->get_qtag() && !pit->get_qid())
96 out_edge(*pit);
97 }
98 out_ << "\n// backward right links\n";
99 out_ << "edge[tailport = se]\n";

103 out_edge(*pit);

129 void out_cluster(word_t &n, const prefix &root) const
130 {
131 const bool clustp =
132 (!root.get_xtag(0) && !root.symbol_limit(0)) ||
133 (!root.get_xtag(1) && !root.symbol_limit(1));
134 if (clustp)
135 out_ << "subgraph cluster_" << n++ << "\n"
136 << "{ style = \"dashed, rounded, setlinewidth(0.2)\"; ";
137 std::vector<prefix> children;
138 for (prefix cur(root); ; )
139 {
140 if (!cur.get_xtag(0) && cur.symbol_limit(0))
141 {
142 prefix pref(cur);
143 pref.go_xlink(0);
144 children.push_back(pref);
145 }
146 if (!cur.get_xtag(1) && cur.symbol_limit(1))
147 {
148 prefix pref(cur);
149 pref.go_xlink(1);
150 children.push_back(pref);
151 }
152 if (clustp)
153 {
154 if (root != cur)
155 out_ << "; ";
156 out_ << 'n' << cur.node_uid();
157 }
158 //
159 if (!cur.get_xtag(0) && !cur.symbol_limit(0))
160 cur.go_xlink(0);
161 else if (!cur.get_xtag(1) && !cur.symbol_limit(1))
162 cur.go_xlink(1);
163 else
164 {
165 if (cur == root)
166 break;
167 for (;;)
168 {
169 word_t parent_id;
170 do
171 {
172 parent_id = cur.get_parent_id();
173 cur.go_parent();
174 } while (cur != root && parent_id);
175 if (cur == root)
176 break;
177 if (!cur.get_xtag(1) && !cur.symbol_limit(1))
178 {
179 cur.go_xlink(1);
180 break;
181 }
182 }
183 if (cur == root)
184 break;
185 }
186 }
187 if (clustp)
188 out_ << " }\n";
189 //
190 for (word_t i = 0; i != children.size(); ++i)
191 out_cluster(n, children[i]);
192 }
